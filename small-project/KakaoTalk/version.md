### 폴더구조
```text

├── example/
│   ├── example1.txt           # 카카오톡 대화 예시 데이터
│   ├── 단톡방저장예시1.txt     # 단체 카카오톡 백업 예시
│   └── 카톡저장예시1.txt       # 개인 카카오톡 백업 예시
├── module/
│   ├── engine.js              # Canvas 2D 그래픽 렌더링 모듈
│   ├── interface.js           # UI 컨트롤러 및 이벤트 연동 모듈
│   └── main.js                # 시뮬레이션 및 재생 라이프사이클 엔트리
├── src/
│   ├── default-setting.md    # 기본 환경설정 파일
│   ├── example1.svg          # 기본 뷰어용 SVG 이미지
│   └── frame.xml             # SVG 레이아웃 템플릿 XML
├── index.html                # 메인 UI 인터페이스
├── kakaotalk-style.css       # 그리드 및 UI 스타일시트
└── version.md                # 버전 및 요구사항 이력 관리
```
